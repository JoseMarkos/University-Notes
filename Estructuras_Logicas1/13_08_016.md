## Insert First Function

    public Nodo prev {get; set;}
    public string name {get; set;}
    public Nodo next {get; set;}
}

void InsertFirst (string name)
{
  var node = new Node();
  node.name = name
  
  if (head != null)
  {
    head.prev = node;
    node.next = head;
  }

  head = node;
}```
